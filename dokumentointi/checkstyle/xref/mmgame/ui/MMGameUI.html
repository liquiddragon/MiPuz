216     @Override
217     public void actionPerformed(ActionEvent e) {
218         if (e.getSource() instanceof JButton) {
219             executeGame((JButton) e.getSource());
220         } else if (e.getSource() instanceof Timer) {
221             //System.out.print("Timer action...");
222             if (e.getSource() == blinkersDelay) {
223                 blinkersDelay.stop();
224             }
225             this.mmTimerReceived(new MMTimerEvent(this, 2));
226         } else {
227             gameTileAction((GameTile) e.getSource());
228         }
229     }

231     /**
232      * Helper method to get game running after player presses Go button.
233      *
234      * @param button that was pressed
235      */
236     private void executeGame(JButton button) {
237         engine.startNewGame();
238 
239         lastShownCount = 1;
240         blinkIndex = 0;
241         blinkTiles();
242 
243         button.setVisible(false);
244     }

246     /**
247      * GameTile action handler.
248      *
249      * @param gt GameTile pressed
250      */
251     private void gameTileAction(GameTile gt) {
252         selections.add(Integer.parseInt(gt.getActionCommand()));
253         //System.out.println("\nSelection: " + Integer.parseInt(gt.getActionCommand()));
254         gt.setSelected(false);
255 
256         if (checkSelections() == true) {
257             if (selections.size() == lastShownCount) {
258                 if (lastShownCount < engine.getGameLevel().getCount()) {
259                     startNextRound(false);
260                 } else {
261                     handleGameOver();
262                 }
263             }
264         } else if (selections.size() == lastShownCount) {
265             startNextRound(true);
266         }
267     }

269     /**
270      * Helper method for handling game over.
271      */
272     private void handleGameOver() {
273         JOptionPane.showMessageDialog(gamePanel, "Congratulations", "MMGame", JOptionPane.PLAIN_MESSAGE);
274 
275         MMStateEvent stateEvent = new MMStateEvent(this, MMStates.State.GAME_OVER);
276         mmStateListener.mmStateReceived(stateEvent);
277     }

279     /**
280      * Helper method to start next round or repeat previous one.
281      *
282      * @param repeat if true previous round is repeated
283      */
284     private void startNextRound(boolean repeat) {
285         if (!repeat) {
286             lastShownCount++;
287         }
288         blinkIndex = 0;
289         blinkersDelay.restart();
290         selections.clear();
291     }
292 
293     /**
294      * Helper method for checking if user entered input was correct.
295      *
296      * @return true if input was correct, otherwise false
297      */
298     private boolean checkSelections() {
299         boolean correct = true;
300 
301         List<Integer> numbers = engine.obtainNumbers(lastShownCount);
302 
303         if (selections.size() == numbers.size()) {
304             for (int i = 0; i < lastShownCount; i++) {
305                 if (!selections.get(i).equals(numbers.get(i))) {
306                     correct = false;
307                     break;
308                 }
309             }
310         } else {
311             correct = false;
312         }
313 
314         return correct;
315     }

317     /**
318      * Helper method for creating GameTile.
319      *
320      * @param normalIndex index for normal image
321      * @param selectedIndex index for selected image
322      * @param actionCommand action command string attached to this GameTile
323      * @param keyEvent shortcut key for this GameTile
324      * @return GameTile created
325      */
326     private GameTile createGameTile(int normalIndex, int selectedIndex, String actionCommand, int keyEvent) {
327         ImageIcon otherImage;
328         if (normalIndex == selectedIndex) {
329             otherImage = getGray(images.get(normalIndex));
330         } else {
331             otherImage = images.get(selectedIndex);
332         }
333         GameTile gt = new GameTile(images.get(normalIndex), otherImage, this);
334 
335         gt.setActionCommand(actionCommand);
336         gt.setMnemonic(keyEvent);
337 
338         return gt;
339     }
340 
341     /**
342      * Helper method to blink tiles to visualise them to player.
343      */
344     private void blinkTiles() {
345         List<Integer> numbers = engine.obtainNumbers(lastShownCount);
346 
347         if (blinkIndex < lastShownCount) {
348             GameTile gt = gameButtons.get(GT.values()[numbers.get(blinkIndex)]);
349             //System.out.println("\nBlink: " + numbers.get(blinkIndex));
350             gt.blink();
351             blinkIndex++;
352         }
353     }

355     /**
356      * Timer event handler for GameTile and game GUI events.
357      *
358      * @param event MMTimerEvent
359      */
360     @Override
361     public void mmTimerReceived(MMTimerEvent event) {
362         switch (event.getID()) {
363             case 1:
364                 blinkers.restart();
365                 //System.out.print("Event 1..." + System.currentTimeMillis() + "...");
366                 break;
367 
368             case 2:
369                 if (blinkers.isRunning()) {
370                     blinkers.stop();
371                 }
372                 //System.out.print("Event 2..." + System.currentTimeMillis() + "...");
373                 blinkTiles();
374                 break;
375         }
376     }
