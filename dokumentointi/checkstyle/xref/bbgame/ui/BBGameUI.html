95      /**
96       * Construct the main display from its sub-components.
97       */
98      private void createGameUI() {
99          gamePanel.setLayout(new BorderLayout());
100 
101         long newGuess = engine.obtainNewGuess();
102         currentGuess = 0;
103 
104         gamePanel.add(createTopPanel(newGuess), BorderLayout.PAGE_START);
105         gamePanel.add(createBBPanel(), BorderLayout.CENTER);
106         gamePanel.add(createBottomPanel(), BorderLayout.PAGE_END);
107 
108     }

110     /**

152     /**
153      * Construct main display button objects.
154      *
155      * @return panel containing main display buttons
156      */
157     private JPanel createBBPanel() {
158         JPanel bbPanel = new JPanel(new GridBagLayout());
159         bbToggles = new ArrayList<>();
160 
161         int column = 0;
162         int row = 0;
163         int keyIndex = 0;
164         int i = engine.getGameLevel().getBits();
165         while (i > 0) {
166             JToggleButton gameButton = createGameButton(i--, keyIndex++);
167             bbToggles.add(gameButton);
168             bbPanel.add(gameButton, new GBC(column, row).setFill(GBC.HORIZONTAL).setWeight(1.0, 1.0));
169 
170             column++;
171             if (column % rowLength == 0) {
172                 row++;
173                 column = 0;
174             }
175         }
176         uiPanels.put(panelMainID, bbPanel);
177 
178         return bbPanel;
179     }

198     /**
199      * Handler for the main game GUI actions.
200      *
201      * @param event to be handled
202      */
203     @Override
204     public void actionPerformed(ActionEvent event) {
205         if (event.getActionCommand().startsWith(bbID) == true) {
206             handleGameCommands(event);
207         } else {
208             switch (event.getActionCommand()) {
209                 case gameOverID:
210                     BBStateEvent stateEvent = new BBStateEvent(this, BBStates.State.GAME_OVER);
211                     bbStateListener.bbStateReceived(stateEvent);
212                     break;
213 
214                 case showHelpID:
215                     handleHelpCommand();
216                     break;
217             }
218         }
219     }
220 
221     /**
222      * Helper method to handle help command and displaying hints.
223      */
224     private void handleHelpCommand() {
225         hintsOn = !hintsOn;
226         if (hintsOn) {
227             showHelp.setText(tCurrent + currentGuess);
228         } else {
229             showHelp.setText(tHelp);
230         }
231     }
232 
233     /**
234      * Helper method for handling game related commands and updating display
235      * accordingly.
236      *
237      * @param event to be handled
238      */
239     private void handleGameCommands(ActionEvent event) {
240         udpateDisplay(event);
241 
242         String bbGuess = obtainGuess(event);
243 
244         if (engine.checkGuess(bbGuess) == BBEngine.CheckResult.CORRECT) {
245             displayGameOver();
246         }
247     }
248 
249     /**
250      * Updates main display buttons based on event.
251      *
252      * @param event to be handled
253      */
254     private void udpateDisplay(ActionEvent event) {
255         for (JToggleButton bbToggle : bbToggles) {
256             if (bbToggle.getActionCommand().equals(event.getActionCommand())) {
257                 long value = obtainButtonValue(bbToggle);
258 
259                 if (bbToggle.isSelected()) {
260                     currentGuess += value;
261                     bbToggle.setText(tBBOne);
262                 } else {
263                     currentGuess -= value;
264                     bbToggle.setText(tBBZero);
265                 }
266 
267                 if (hintsOn) {
268                     showHelp.setText(tCurrent + currentGuess);
269                 }
270             }
271         }
272     }

274     /**
275      * Helper method for obtaining value based on selected button.
276      *
277      * @param bbToggle button pressed
278      * @return value for selected button
279      */
280     private long obtainButtonValue(JToggleButton bbToggle) {
281         String buttonValue;
282 
283         buttonValue = new String(bbToggle.getActionCommand().substring(bbID.length()));
284 
285         long value = 0;
286         try {
287             value = Integer.parseInt(buttonValue);
288             value = 1L << (value - 1L);
289         } catch (NumberFormatException e) {
290             System.out.println("Internal error: " + e.toString());
291         }
292 
293         return value;
294     }

296     /**
297      * Helper method for obtaining a string formed by buttons.
298      *
299      * @param event to be handled
300      * @return String containing user's guess
301      */
302     private String obtainGuess(ActionEvent event) {
303         StringBuilder bbGuess = new StringBuilder();
304         for (JToggleButton bbToggle : bbToggles) {
305             if (bbToggle.isSelected()) {
306                 bbGuess.append(tBBOne);
307             } else {
308                 bbGuess.append(tBBZero);
309             }
310         }
311         return bbGuess.toString();
312     }

314     /**
315      * Helper method for handling Game Over message displaying and related
316      * action creation.
317      */
318     private void displayGameOver() {
319         JPanel bottomPanel = uiPanels.get(panelBottomID);
320         bottomPanel.removeAll();
321 
322         bottomPanel.add(Box.createGlue(), new Float(1));
323 
324         JButton gameOver = new JButton(tGameOver);
325         gameOver.setActionCommand(gameOverID);
326         gameOver.setMnemonic(KeyEvent.VK_O);
327         gameOver.addActionListener(this);
328 
329         bottomPanel.add(gameOver, (float) 0.4);
330         bottomPanel.add(Box.createGlue(), new Float(1));
331 
332         udpateMainGamePanel();
333     }

335     /**
336      * Request update to the main game panel.
337      */
338     private void udpateMainGamePanel() {
339         gamePanel.revalidate();
340         gamePanel.repaint();
341     }
